关键字与操作符

	-as
	-as?
	-break
	-class
	-continue
	-do
	-else
	-false
	-for
	-fun
	-if
	-in

	-is

	-null

	-return

	-this
	-throw
	-true
	-try
	-typealias
	-val
	-var
	-when
	-while

	-catch
	-constructor
	-delegate
	-dynamic
	-field
	-file
	-finally
	-get
	-import
	-init
	-param
	-property
	-receiver
	-set
	-setparam
	-where

	-annotation
	-companion
	-const
	-crossinline

	-enum
	-expect
	-external

	-open
	-operator
	-out
	-override
	-private
	-protected
	-public
	-reified
	-sealed
	-suspend
	-tailrec
	-vararg

	-it 用在lambda表达式内部来隐式引用其参数
